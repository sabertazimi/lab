295fn spawn_watch_shell(
296    failed_exercise_hint: &Arc<Mutex<Option<String>>>,
297    should_quit: Arc<AtomicBool>,
298) {
299    let failed_exercise_hint = Arc::clone(failed_exercise_hint);
300    println!("Welcome to watch mode! You can type 'help' to get an overview of the commands you can use here.");
301    thread::spawn(move || loop {
302        let mut input = String::new();
303        match io::stdin().read_line(&mut input) {
304            Ok(_) => {
305                let input = input.trim();
306                if input == "hint" {
307                    if let Some(hint) = &*failed_exercise_hint.lock().unwrap() {
308                        println!("{hint}");
309                    }
310                } else if input == "clear" {
311                    println!("\x1B[2J\x1B[1;1H");
312                } else if input.eq("quit") {
313                    should_quit.store(true, Ordering::SeqCst);
314                    println!("Bye!");
315                } else if input.eq("help") {
316                    println!("Commands available to you in watch mode:");
317                    println!("  hint  - prints the current exercise's hint");
318                    println!("  clear - clears the screen");
319                    println!("  quit  - quits watch mode");
320                    println!("  help  - displays this help message");
321                    println!();
322                    println!("Watch mode automatically re-evaluates the current exercise");
323                    println!("when you edit a file's contents.")
324                } else {
325                    println!("unknown command: {input}");
326                }
327            }
328            Err(error) => println!("error reading command: {error}"),
329        }
330    });
331}

333fn find_exercise<'a>(name: &str, exercises: &'a [Exercise]) -> &'a Exercise {
334    if name.eq("next") {
335        exercises
336            .iter()
337            .find(|e| !e.looks_done())
338            .unwrap_or_else(|| {
339                println!("ðŸŽ‰ Congratulations! You have done all the exercises!");
340                println!("ðŸ”š There are no more exercises to do next!");
341                std::process::exit(1)
342            })
343    } else {
344        exercises
345            .iter()
346            .find(|e| e.name == name)
347            .unwrap_or_else(|| {
348                println!("No exercise found for '{name}'!");
349                std::process::exit(1)
350            })
351    }
352}

354enum WatchStatus {
355    Finished,
356    Unfinished,
357}

359fn watch(exercises: &[Exercise], verbose: bool) -> notify::Result<WatchStatus> {
360    /* Clears the terminal with an ANSI escape code.
361    Works in UNIX and newer Windows terminals. */
362    fn clear_screen() {
363        println!("\x1Bc");
364    }
365
366    let (tx, rx) = channel();
367    let should_quit = Arc::new(AtomicBool::new(false));
368
369    let mut watcher: RecommendedWatcher = Watcher::new(tx, Config::default())?;
370    watcher.watch(Path::new("./exercises"), RecursiveMode::Recursive)?;
371
372    clear_screen();
373
374    let to_owned_hint = |t: &Exercise| t.hint.to_owned();
375    let failed_exercise_hint = match verify(exercises.iter(), verbose) {
376        Ok(_) => return Ok(WatchStatus::Finished),
377        Err(exercise) => Arc::new(Mutex::new(Some(to_owned_hint(exercise)))),
378    };
379    spawn_watch_shell(&failed_exercise_hint, Arc::clone(&should_quit));
380    loop {
381        match rx.recv_timeout(Duration::from_secs(1)) {
382            Ok(Ok(event)) => {
383                // Check for Create, Modify (Write), or Metadata change events
384                if matches!(
385                    event.kind,
386                    EventKind::Create(_)
387                        | EventKind::Modify(notify::event::ModifyKind::Data(
388                            notify::event::DataChange::Any
389                        ))
390                        | EventKind::Modify(notify::event::ModifyKind::Metadata(_))
391                ) {
392                    for path in event.paths {
393                        if path.extension() == Some(OsStr::new("rs")) && path.exists() {
394                            let filepath = path.canonicalize().unwrap();
395                            let pending_exercises =
396                                exercises
397                                    .iter()
398                                    .skip_while(|e| !filepath.ends_with(&e.path))
399                                    // .filter(|e| filepath.ends_with(&e.path))
400                                    .chain(exercises.iter().filter(|e| {
401                                        !e.looks_done() && !filepath.ends_with(&e.path)
402                                    }));
403                            clear_screen();
404                            match verify(pending_exercises, verbose) {
405                                Ok(_) => return Ok(WatchStatus::Finished),
406                                Err(exercise) => {
407                                    let mut failed_exercise_hint =
408                                        failed_exercise_hint.lock().unwrap();
409                                    *failed_exercise_hint = Some(to_owned_hint(exercise));
410                                }
411                            }
412                            break; // Only handle the first .rs file path
413                        }
414                    }
415                }
416            }
417            Err(RecvTimeoutError::Timeout) => {
418                // the timeout expired, just check the `should_quit` variable below then loop again
419            }
420            Ok(Err(e)) => println!("watch error: {e:?}"),
421            Err(e) => println!("watch error: {e:?}"),
422        }
423        // Check if we need to exit
424        if should_quit.load(Ordering::SeqCst) {
425            return Ok(WatchStatus::Unfinished);
426        }
427    }
428}
